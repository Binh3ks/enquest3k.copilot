export default function FreeTalkTab() {
  const [messages, setMessages] = useState([
    { 
      role: 'assistant', 
      text: "Hello! üëã I am happy to see you! How are you today?",
      mood: "üëã",
      hints: ["I am happy!", "I am tired.", "I am hungry."] 
    }
  ]);
  const [inputText, setInputText] = useState("");
  const [isRecording, setIsRecording] = useState(false);
  const [isLoading, setIsLoading] = useState(false);
  
  const messagesEndRef = useRef(null);

  // Auto-scroll
  useEffect(() => {
    messagesEndRef.current?.scrollIntoView({ behavior: "smooth" });
  }, [messages]);

  const handleSend = async (textOverride = null) => {
    const textToSend = textOverride || inputText;
    if (!textToSend.trim()) return;

    // Add User Message
    const userMsg = { role: 'user', text: textToSend };
    setMessages(prev => [...prev, userMsg]);
    setInputText("");
    setIsLoading(true);

    try {
      // Call AI
      const aiResponse = await mockFreeTalkApi(messages, textToSend);
      
      // Add AI Message with Scaffolding Data
      setMessages(prev => [...prev, { 
        role: 'assistant', 
        text: aiResponse.text,
        mood: aiResponse.mood, // Emoji mood
        hints: aiResponse.hints // Suggested replies
      }]);
    } catch (error) {
      console.error("Chat Error", error);
    } finally {
      setIsLoading(false);
    }
  };

  const handleMicClick = () => {
    setIsRecording(!isRecording);
    // Simulation: Voice input logic
    if (!isRecording) {
      setTimeout(() => {
        setIsRecording(false);
        setInputText("I played football."); 
      }, 1500);
    }
  };

  const handleHintClick = (hintText) => {
    handleSend(hintText);

  return (

               üë©‚Äçüè´

        {messages.map((msg, idx) => (
          <div key={idx} className={`flex flex-col w-full ${msg.role === 'user' ? 'items-end' : 'items-start'}`}>
            
            <div className={`
              max-w-[85%] px-5 py-4 rounded-3xl text-lg md:text-xl leading-relaxed shadow-sm relative transition-all duration-300
              ${msg.role === 'user' 
                ? 'bg-indigo-600 text-white rounded-br-none' 
                : 'bg-white text-slate-800 border border-slate-100 rounded-bl-none'}
            `}>
              {/* Visual Cue (Mood Emoji) for AI */}
              {msg.role === 'assistant' && msg.mood && (
                <div className="absolute -top-4 -left-2 text-3xl filter drop-shadow-md animate-bounce-slow">
                  {msg.mood}
                </div>
              )}
              
              <div className="mt-1">{msg.text}</div>

              {/* TTS Button */}
              {msg.role === 'assistant' && (
                <button className="absolute -bottom-3 -right-2 bg-indigo-50 text-indigo-600 p-1.5 rounded-full shadow-sm hover:scale-110 transition-transform">
                  <Volume2 size={16} />
                </button>
              )}
            </div>

            {/* SCAFFOLDING: REPLY CHIPS (Only show for the latest assistant message) */}
            {msg.role === 'assistant' && idx === messages.length - 1 && msg.hints && (
              <div className="mt-3 flex flex-wrap gap-2 animate-fade-in-up ml-2">

                    disabled={!inputText.trim()}
